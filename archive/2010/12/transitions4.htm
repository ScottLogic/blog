JavaScript 
$(document).ready(function() {
  $("#insertRow").click(function() {
    var tbl = document.getElementById('tbl'),
      newRow = tbl.rows[0].cloneNode(true);
    tbl.rows[0].parentNode.insertBefore(newRow, tbl.rows[0]);
  });
});
